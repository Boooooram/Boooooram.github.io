顺序表 - 类模版

//
//  SeqList.cpp
//  cpp
//
//  Created by xuxiaokang on 2018/11/28.

//

#include <iostream>
using namespace std;

#define DEFAULT_SIZE 1000
enum Status {SUCCESS, FAIL, UNDER_FLOW, OVER_FLOW,RANGE_ERROR, DUPLICATE_ERROR, NOT_PRESENT, ENTRY_INSERTED, ENTRY_FOUND, VISITED, UNVISITED};

template <typename T>
class SeqList
{
private:
    T *elems;                                           // 元素存储空间的首地址
    int length;                                         // 顺序表的当前长度
    int maxLength;                                      // 顺序表的最大容量

public:
    SeqList(int size = DEFAULT_SIZE);                   // 构造一个空表
    SeqList(T v[], int n, int size = DEFAULT_SIZE);     // 根据数组v构建长度为n的顺序表
    virtual ~SeqList();                                 // 析构函数
    int getLength() const;                              // 返回当前顺序表长度
    int getMaxLength() const;                           // 返回当前顺序表最大容量
    bool isEmpty() const;                               // 判断顺序表是否为空
    void clear();                                       // 清空顺序表
    void traverse() const;                              // 遍历顺序表
    int LocateElem(const T &e) const;                   // 返回指定元素的下标
    Status GetElem(int i, T &e) const;                  // 取顺序表中第i个元素的值
    Status SetElem(int i, const T &e);                  // 修改顺序表中第i个元素的值
    Status DeleteElem(int i, T &e);                     // 删除顺序表中第i个元素
    Status InsertElem(int i, const T &e);               // 在顺序表第i个位置插入元素
    Status InsertElem(const T &e);                      // 在顺序表末尾插入元素
    SeqList(const SeqList<T> &sa);                      // 复制构造函数
    SeqList<T>& operator= (const SeqList<T> &sa);       // 赋值语句重载
};


template <typename T>
SeqList<T>::SeqList(int size) {
    elems = new T[size];
    length = 0;
    maxLength = size;
}

template <typename T>
SeqList<T>::SeqList(T v[], int n, int size) {
    elems = new T[size];
    length = n;
    maxLength = size;
    for(int i = 0; i < n; i++)
        elems[i] = v[i];
}

template <typename T>
SeqList<T>::~SeqList() {
    delete []elems;
}

template <typename T>
int SeqList<T>::getLength() const {
    return length;
}

template <typename T>
int SeqList<T>::getMaxLength() const {
    return maxLength;
}

template <typename T>
bool SeqList<T>::isEmpty() const {
    return length == 0;
}

template <typename T>
void SeqList<T>::clear() {
    length = 0;
}

template <typename T>
void SeqList<T>::traverse() const {
    cout << "[";
    for(int i = 0; i < length; i++) {
        if(i == length - 1)
            cout << elems[i];
        else
            cout << elems[i] << ", ";
    }
    cout << "]" << endl;
}

template <typename T>
int SeqList<T>::LocateElem(const T &e) const {
    int locat = -1;
    for(int i = 0; i < length; i++) {
        if(elems[i] == e)
            locat = i + 1;
    }
    return locat;
}

template <typename T>
Status SeqList<T>::GetElem(int i, T &e) const {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        e = elems[i - 1];
        return ENTRY_FOUND;
    }
}

template <typename T>
Status SeqList<T>::SetElem(int i, const T &e) {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        elems[i - 1] = e;
        return SUCCESS;
    }
}

template<typename T>
Status SeqList<T>::DeleteElem(int i, T &e) {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        e = elems[i -1];
        for(int j = i; j < length; j++)
            elems[j - 1] = elems[j];

        length -= 1;
        return SUCCESS;
    }
}

template<typename T>
Status SeqList<T>::InsertElem(int i, const T &e) {
    if(length == maxLength)
        return OVER_FLOW;
    else if(i < 1 || i > length + 1)
        return RANGE_ERROR;
    else {
        for(int j = length; j >= i; j--)
            elems[j + 1] = elems[j];
        elems[i - 1] = e;
        length += 1;
        return SUCCESS;

    }
}
template<typename T>
Status SeqList<T>::InsertElem(const T &e) {
    if(length == maxLength)
        return OVER_FLOW;
    else {
        elems[length] = e;
        length += 1;
        return SUCCESS;
    }
}

template<typename T>
SeqList<T>::SeqList(const SeqList<T> &sa) {
    int salength = sa.getLength();
    maxLength = sa.getMaxLength();
    elems = new T[maxLength];
    length = 0;
    T e;
    for(int i = 1; i <= salength; i++) {
        sa.GetElem(i, e);
        this->InsertElem(e);
    }
}

template<typename T>
SeqList<T>& SeqList<T>::operator= (const SeqList<T> &sa) {
    if(&sa != this) {
        int salength = sa.getLength();
        int maxLength = sa.getMaxLength();
        elems = new T[maxLength];
        length = 0;
        T e;
        for(int i = 1; i <= salength; i++) {
            sa.GetElem(i, e);
            this->InsertElem(e);
        }
    }
    return *this;
}

顺序表 - 应用

int main() {
    int a[5] = {1, 2, 3, 4, 5};
    SeqList<int> sa(a, 5, 10);

    cout << "顺序表sa的当前长度为：" << sa.getLength() << endl;
    cout << "顺序表sa的容量为：" << sa.getMaxLength() << endl;
    cout << "顺序表sa是否为空？：" << (sa.isEmpty() ? "是" : "否") << endl;

    const int d = 5;
    cout << "元素5的下标为：" << sa.LocateElem(d) << endl;

    SeqList<int> sb(sa);
    cout << "通过复制构造得到的顺序表sb的遍历结果为：";
    sb.traverse();

    int e;
    const int f = 8;
    sb.GetElem(3, e);
    cout << "sb的第3个元素为：" << e << endl;

    sb.SetElem(3, f);
    cout << "sb经过修改元素后的遍历结果为：";
    sb.traverse();

    sb.DeleteElem(3, e);
    cout << "sb经过删除元素后的遍历结果为：";
    sb.traverse();
    cout << "顺序表sb的当前长度为：" << sb.getLength() << endl;
    cout << "e的值为：" << e << endl;

    e = 20;
    sb.InsertElem(3, e);
    cout << "sb经过插入元素后的遍历结果为：";
    sb.traverse();
    cout << "顺序表sb的当前长度为：" << sb.getLength() << endl;

    sb.InsertElem(88);
    cout << "sb经过插入元素后的遍历结果为：";
    sb.traverse();
    cout << "顺序表sb的当前长度为：" << sb.getLength() << endl;

    cout << "顺序表sa遍历结果为：";
    sa.traverse();

    sa.clear();
    cout << "顺序表清空后，遍历结果为：";
    sa.traverse();

    SeqList<int> sc;
    sc = sb;
    cout << "顺序表sc遍历结果为：";
    sc.traverse();
    return 0;
}

// 定义1
struct Student {
    int a;
};
// 声明(2种)
struct Student stu1;
Student stu2;

// 定义2(单个别名)
typedef struct Student {
    int a;
} Stu;
// 即 Stu = struct Student

// 声明(3种)
struct Student stu1;
Student stu2;
Stu stu3;

// 定义3(多个别名)
typedef struct Student {
    int a;
} Stu1, Stu2, Stu3;
// 声明
Stu1 stu1;
Stu2 stu2;
Stu3 stu3;

struct Student {
    int a;

struct Student {

ps -ef | grep mysql

./mysql.server stop

./mysqld_safe --skip-grant-tables &

进入mysql交互模式

mysql

选择【mysql】数据库

use mysql

刷新mysql的系统权限相关表

flush privileges;

ALTER USER 'root'@'localhost' IDENTIFIED BY '新密码';

注意：重启mysql

./mysql.server stop
./mysql.server start

修改”Name”为”Tomcat 9.0.7”(强迫症)

点击”+”, 选择”Artifact”

点击”Apply”&”OK”

运行”HelloWorld”

在”src”目录中添加”HelloWorld.java”文件，并添加如下代码:

// 导入必需的 java 库
import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

// 扩展 HttpServlet 类
public class HelloWorld extends HttpServlet {

    private String message;

    public void init() throws ServletException
    {
        // 执行必需的初始化
        message = "Hello World";
    }

    public void doGet(HttpServletRequest request,
                    HttpServletResponse response)
            throws ServletException, IOException
    {
        // 设置响应内容类型
        response.setContentType("text/html");

        // 实际的逻辑是在这里
        PrintWriter out = response.getWriter();
        out.println("<h1>" + message + "</h1>");
    }

    public void destroy()
    {
        // 什么也不做
    }
}

在 web - WEB-INF - web.xml 中添加如下代码:

<servlet>
    <servlet-name>HelloWorld</servlet-name>
    <servlet-class>HelloWorld</servlet-class>
</servlet>

<servlet-mapping>
    <servlet-name>HelloWorld</servlet-name>
    <url-pattern>/HelloWorld</url-pattern>
</servlet-mapping>

运行tomcat
