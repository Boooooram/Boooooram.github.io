linux - update gcc

发表于 2019-08-11 | 更新于 2019-08-12 | 分类于 linux

1.下载源码（链接）

找到想要的版本，要么用wget下载，要么下载到本地后上传到服务器

wget http://ftp.gnu.org/gnu/gcc/gcc-9.1.0/gcc-9.1.0.tar.gz

2.解压

解压到你希望的目录，我解压到了当前目录

tar -zxvf gcc-9.1.0.tar.gz

3.安装依赖

cd gcc-9.1.0
./contrib/download_prerequisites

4.建立临时目录，存放编译过程中的中间文件

mkdir gcc-9.1.0-build
cd gcc-9.1.0-build
../configure --disable-multilib

不加『 –disable-multilib 』会出现以下错误信息：
configure: error: I suspect your system does not have 32-bit development libraries (libc and headers). If you have them, rerun configure with –enable-multilib. If you do not have them, and want to build a 64-bit-only compiler, rerun configure with –disable-multilib.

5.编译 & 安装

单核cpu编译太久了，挂在后台干其他事情去了，cpu核数多的话，用 -j 选项

nohup make &

出现『 make[1]: Leaving directory `/Downloads/gcc-9.1.0/gcc-9.1.0-build’ 』就说明编译好了，编译好之后，再安装，我还是选择挂在后台安装

nohup make install &

因为运行configure命令时，没有指定安装目录，因此上述命令会将最新版本的gcc编译器安装到默认位置：/usr/local，也就是说，头文件在/usr/local/include目录，可执行文件在/usr/local/bin目录，库文件在/usr/local/lib目录。

6.指定本机使用最新版本gcc编译器

使用『 lternatives 』命令配置增加最新版本编译器，注意：gcc是编译c程序的默认编译器，g++是编译c++程序的默认编译器。

alternatives --install /usr/bin/gcc gcc /usr/local/bin/gcc 50
alternatives --install /usr/bin/g++ g++ /usr/local/bin/g++ 50

用法：alternatives --install <link> <name> <path> <priority>
                [--initscript <service>]
                [--slave <link> <name> <path>]*
    alternatives --remove <name> <path>
    alternatives --auto <name>
    alternatives --config <name>
    alternatives --display <name>
    alternatives --set <name> <path>

选择默认使用的GCC编译器版本：

alternatives --config gcc

7.查看gcc和g++版本

gcc -v
g++ -v

8.补充

8.1 确认默认语言标准

gcc-3.0.4 gcc-3.1.1 gcc-3.2.3 gcc-3.3.6 gcc-3.4.6 gcc-4.0.4 gcc-4.1.2 gcc-4.2.4 默认是 -std=gnu89    
gcc-4.3.6 gcc-4.4.7 gcc-4.5.4 gcc-4.6.4 gcc-4.7.4 gcc-4.8.5 gcc-4.9.4 gcc-5.4.0 默认是 -std=gnu++98
gcc-6.4.0 gcc-7.2.0 默认是 -std=gnu++14

8.2 编译 cpp 文件（gcc 版本 9.1.0）

g++ -pthread demo.cpp -o demo

8.3 运行文件时，出现『 /usr/lib64/libstdc++.so.6: version `GLIBCXX_3.4.22’ not found 』

原因：升级gcc时，生成的动态库没有替换老版本gcc的动态库导致的，将gcc最新版本的动态库替换系统中老版本的动态库即可解决。

检查动态库

strings /usr/lib64/libstdc++.so.6 | grep GLIBC

查找编译gcc时生成的最新动态库

find / -name "libstdc++.so*"

将查找到的最新动态库libstdc++.so.x.x.xx复制到/usr/lib64目录下

cp /Downloads/gcc-9.1.0/gcc-9.1.0-build/stage1-x86_64-pc-linux-gnu/libstdc++-v3/src/.libs/libstdc++.so.6.0.26 /usr/lib64

删除软链接

cd /usr/lib64
rm -rf libstdc++.so.6

将默认库的软连接指向最新动态库

ln -s libstdc++.so.6.0.26 libstdc++.so.6

默认动态库升级完成。重新运行以下命令检查动态库：

strings /usr/lib64/libstdc++.so.6 | grep GLIBC

stride = 1 时

设input = [M, M], kernel_size = [F, F], stride = [1, 1]

padding = valid 时

output = ( M - F ) / 1 + 1

padding = same 时

P = ( F - 1 ) / 2
output = ( M + 2*P - F ) / 1 + 1

stride != 1 时

设input = [M, M], kernel_size = [F, F], stride = [s, s]

padding = valid 时

output = ceil( ( M - F + 1 ) / s )

padding = same 时

output = ceil( M / s )

template<typename T>
void permutation(T list[], int p, int q) {
    if(p == q) {
        for(int i = 0; i <= q; i++)
            cout << list[i];
        cout << endl;
    }
    else {
        for(int i = p; i <= q; i++) {
            swap(list[i], list[p]);
            perm(list, p + 1, q);
            swap(list[i], list[p]);
        }
    }
}

template<typename T>
inline void swap(T &a, T &b) {
    T temp = a;
    a = b;
    b = temp;
}

//

#include <iostream>
using namespace std;

#define DEFAULT_SIZE 1000
enum Status {SUCCESS, FAIL, UNDER_FLOW, OVER_FLOW,RANGE_ERROR, DUPLICATE_ERROR, NOT_PRESENT, ENTRY_INSERTED, ENTRY_FOUND, VISITED, UNVISITED};

template <typename T>
class SeqList
{
private:
    T *elems;                                           // 元素存储空间的首地址
    int length;                                         // 顺序表的当前长度
    int maxLength;                                      // 顺序表的最大容量

public:
    SeqList(int size = DEFAULT_SIZE);                   // 构造一个空表
    SeqList(T v[], int n, int size = DEFAULT_SIZE);     // 根据数组v构建长度为n的顺序表
    virtual ~SeqList();                                 // 析构函数
    int getLength() const;                              // 返回当前顺序表长度
    int getMaxLength() const;                           // 返回当前顺序表最大容量
    bool isEmpty() const;                               // 判断顺序表是否为空
    void clear();                                       // 清空顺序表
    void traverse() const;                              // 遍历顺序表
    int LocateElem(const T &e) const;                   // 返回指定元素的下标
    Status GetElem(int i, T &e) const;                  // 取顺序表中第i个元素的值
    Status SetElem(int i, const T &e);                  // 修改顺序表中第i个元素的值
    Status DeleteElem(int i, T &e);                     // 删除顺序表中第i个元素
    Status InsertElem(int i, const T &e);               // 在顺序表第i个位置插入元素
    Status InsertElem(const T &e);                      // 在顺序表末尾插入元素
    SeqList(const SeqList<T> &sa);                      // 复制构造函数
    SeqList<T>& operator= (const SeqList<T> &sa);       // 赋值语句重载
};


template <typename T>
SeqList<T>::SeqList(int size) {
    elems = new T[size];
    length = 0;
    maxLength = size;
}

template <typename T>
SeqList<T>::SeqList(T v[], int n, int size) {
    elems = new T[size];
    length = n;
    maxLength = size;
    for(int i = 0; i < n; i++)
        elems[i] = v[i];
}

template <typename T>
SeqList<T>::~SeqList() {
    delete []elems;
}

template <typename T>
int SeqList<T>::getLength() const {
    return length;
}

template <typename T>
int SeqList<T>::getMaxLength() const {
    return maxLength;
}

template <typename T>
bool SeqList<T>::isEmpty() const {
    return length == 0;
}

template <typename T>
void SeqList<T>::clear() {
    length = 0;
}

template <typename T>
void SeqList<T>::traverse() const {
    cout << "[";
    for(int i = 0; i < length; i++) {
        if(i == length - 1)
            cout << elems[i];
        else
            cout << elems[i] << ", ";
    }
    cout << "]" << endl;
}

template <typename T>
int SeqList<T>::LocateElem(const T &e) const {
    int locat = -1;
    for(int i = 0; i < length; i++) {
        if(elems[i] == e)
            locat = i + 1;
    }
    return locat;
}

template <typename T>
Status SeqList<T>::GetElem(int i, T &e) const {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        e = elems[i - 1];
        return ENTRY_FOUND;
    }
}

template <typename T>
Status SeqList<T>::SetElem(int i, const T &e) {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        elems[i - 1] = e;
        return SUCCESS;
    }
}

template<typename T>
Status SeqList<T>::DeleteElem(int i, T &e) {
    if(i < 1 || i > length)
        return RANGE_ERROR;
    else {
        e = elems[i -1];
        for(int j = i; j < length; j++)
            elems[j - 1] = elems[j];

        length -= 1;
        return SUCCESS;
    }
}

template<typename T>
Status SeqList<T>::InsertElem(int i, const T &e) {
    if(length == maxLength)
        return OVER_FLOW;
    else if(i < 1 || i > length + 1)
        return RANGE_ERROR;
    else {
        for(int j = length; j >= i; j--)
            elems[j + 1] = elems[j];
        elems[i - 1] = e;
        length += 1;
        return SUCCESS;

    }
}
template<typename T>
Status SeqList<T>::InsertElem(const T &e) {
    if(length == maxLength)
        return OVER_FLOW;
    else {
        elems[length] = e;
        length += 1;
        return SUCCESS;
    }
}

template<typename T>
SeqList<T>::SeqList(const SeqList<T> &sa) {
    int salength = sa.getLength();
    maxLength = sa.getMaxLength();
    elems = new T[maxLength];
    length = 0;
    T e;
    for(int i = 1; i <= salength; i++) {
        sa.GetElem(i, e);
        this->InsertElem(e);
    }
}

template<typename T>
SeqList<T>& SeqList<T>::operator= (const SeqList<T> &sa) {
    if(&sa != this) {
        int salength = sa.getLength();
        int maxLength = sa.getMaxLength();
        elems = new T[maxLength];
        length = 0;
        T e;
        for(int i = 1; i <= salength; i++) {
            sa.GetElem(i, e);
            this->InsertElem(e);
        }
    }
    return *this;
}
