1.macOS 下的 mysql 服务命令

1.1 启动服务

mysql.server start

1.2 停止服务

mysql.server stop

1.3 检查服务

mysql.server status

1.4 重启服务

mysql.server restart

1.5 重载服务

mysql.server reload

2. database 的 CURD

2.1 增 - C

2.1.1 直接创建

create database db_name;

2.1.2 使用 utf8 字符集

create database db_name character set utf8;

2.1.3 对存入数据的编码进行检查

create database db_name character set utf8 collate utf8_general_ci;

2.2 改 - U

2.2.1 无法修改 db_name

2.2.2 修改数据库字符集(charset)

alter database db_name character set new_charset;

2.3 查 - R

2.3.1 查看数据库

show databases;

2.3.2 查看创建数据库的语句信息

show create database db_name;

2.4 删 - D

2.4.1

drop database db_name;

3. table 的 CURD

3.1 增 - C

3.1.1 创建表(A table must have at least 1 column)

create table table_name（
    column_name column_type [NOT NULL | NULL] [DEFAULT default_value]
    [AUTO_INCREMENT] [UNIQUE [KEY]] [[PRIMARY] KEY][COMMENT 'string'])

3.2 改 - U

3.2.1 修改 table_name

rename table old_name to new_name;

3.2.2 增加一个字段

alter table table_name add column column_name column_type;

3.2.3 修改一个字段

alter table table_name modify column column_name new_column_type;

3.2.4 删除一个字段

alter table table_name drop column column_name;

3.2.5 修改表的字符集

alter table table_name character set new_charset;

3.3 查 - R

3.3.1 查看表

show tables;

3.3.2 查看创建表的语句信息

show create table table_name;

3.3.3 查看表结构

desc table_name;

3.4 删 - D

3.4.1 删除表

drop table table_name;

4 data 的 CURD

4.1 增 - C

4.1.1 包含所有 column

insert into table_name values(val1, val2, val3, ... valn);

4.1.2 包含部分 column

insert into table_name(col1, col2, coln) values(val1, val2, valn);

4.2 改 - U

4.2.1 修改某一行的字段

update table_name set column_name = new_value;

4.3 查 - R

4.3.1 表格形式查所有信息

select * from table_name

4.3.2 分块形式查看所有信息

select * from table_name \G

4.4 删 - D

4.4.1 根据条件删除 record

delete from table_name where [conditions];

4.4.2 删除全部 record

delete from table_name;
or
truncate table table_name;
